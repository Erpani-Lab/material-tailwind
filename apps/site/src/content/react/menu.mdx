# Tailwind CSS Menu - React

Easily create `Menu` with different statuses and sizes using our components based on Tailwind CSS and React.

`Menu` fields are an essential user interface design element, providing users with the means to enter non-standardized responses.

See below our `Menu` component examples. It comes in different styles and colors, so you can adapt it easily to your needs.

---

## Nav Menu

<ComponentPreview demo={<Menu.NavMenu />}>
```tsx
import * as React from "react";
import { Menu, Button, Card, Typography } from "@material-tailwind/react";
import { NavArrowDown, Rocket } from "iconoir-react";

const NavItem = React.forwardRef(
  ({ title, description, ...rest }, ref) => {
    return (
      <Menu.Item ref={ref} {...rest} className="flex-col items-start">
        <Typography as="h6" className="font-semibold">
          {title}
        </Typography>
        <Typography
          variant="small"
          className="text-gray-600"
        >
          {description}
        </Typography>
      </Menu.Item>
    );
  },
);

export function NavMenu() {
  return (
    <Menu>
      <Menu.Trigger

        className="flex items-center gap-1"
      >
        Versions{" "}
        <NavArrowDown className="h-4 w-4 group-data-[open=true]:rotate-180" />

        <Card
          color="primary"
          className="col-span-2 grid place-items-center rounded-[5px] px-8 py-4 text-primary-content shadow-none"

          <div>
```

## Notification Menu

<ComponentPreview demo={<Menu.NotificationMenu />}>
```tsx
import * as React from "react";
import {
  Menu,
  IconButton,
  Typography,
  Avatar,
} from "@material-tailwind/react";
import { Bell, Clock } from "iconoir-react";

const MenuItem = React.forwardRef(
  ({ img, title, date, ...rest }, ref) => {
    return (
      <Menu.Item ref={ref} {...rest} className="gap-3">
        <Avatar src={img} alt="profile-picture" />
        <div>
          <Typography variant="small" className="mb-1 block font-semibold">
            {title}
          </Typography>
          <Typography
            variant="small"
            className="flex items-center gap-1 text-sm text-gray-600"
          >
            <Clock className="h-3.5 w-3.5" />
            {date}
          </Typography>
        </div>
      </Menu.Item>
    );
  },
);

export function NotificationMenu() {
  return (
    <Menu>
      <Menu.Trigger as={IconButton} variant="ghost" size="sm">
        <Bell className="h-4 w-4 stroke-2" />
      </Menu.Trigger>
      <Menu.Content>
        <MenuItem
          img="https://dub.sh/iu8bOfU"
          title="Tania send you a message"
          date="13 minutes ago"
        />
        <MenuItem
          img="https://dub.sh/6i238JA"
          title="Natali replied to your email."
          date="1 hour ago"
        />
        <MenuItem
          img="https://dub.sh/iuFsnmW"
          title="You've received a payment."
          date="5 hours ago"
        />
      </Menu.Content>
    </Menu>
  );
}
```
</ComponentPreview>
